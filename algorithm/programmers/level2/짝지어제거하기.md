# 짝지어제거하기

```javascript
function solution(s)
{
    const stack = [];
    for (let c of s) {
        if (stack.length === 0) {
            stack.push(c);
        } else if (stack[stack.length - 1] === c) {
            stack.pop();
        } else {
            stack.push(c);
        }
    }
    return stack.length === 0 ? 1 : 0
}

console.log(solution("baabaa"));
console.log(solution("cdcd"));
console.log(solution("abcdcdab"));
console.log(solution("cdcddcdc"));
console.log(solution(""));
console.log(solution("a"));
console.log(solution("aabb"));
console.log(solution("aaabbb"));
console.log(solution("abccba"));
console.log(solution("aaabba"));

console.log(solution("daabbd"));
console.log(solution("aaaa"));
```
