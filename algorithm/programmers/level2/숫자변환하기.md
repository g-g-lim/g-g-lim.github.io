# 숫자변환하기

```javascript
function solution(x, y, n) {
    let queue = [{ v: y, cnt: 0 }];
    let visited = {};

    while (queue.length > 0) {
        let { v, cnt } = queue.shift();

        if (v === x) {
            return cnt;
        }

        if (v % 3 === 0 && !visited[v / 3]) {
            queue.push({ v: v / 3, cnt: cnt + 1 });
        }
        if (v % 2 === 0 && !visited[v / 2]) {
            queue.push({ v: v / 2, cnt: cnt + 1 });
        }
        if (v - n >= x && !visited[v - n]) {
            queue.push({ v: v - n, cnt: cnt + 1 });
        }
    }

    return -1;
}
```
