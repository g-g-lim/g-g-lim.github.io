# 예상대진표

<img src="/assets/images/programmers/예상대진표.png" width="100%" alt="예상대진표" />

```javascript
function solution(n,a,b){
    let answer = 1;
    const valid = (a, b) => {
        if (a < b) {
            if (b % 2 === 0 && Math.abs(a - b) === 1) {
                return true;
            }
        } else {
            if (a % 2 === 0 && Math.abs(a - b) === 1) {
                return true;
            }
        }
        return false;
    }
    while (!valid(a, b)) {
        a = Math.round(a / 2);
        b = Math.round(b / 2);
        answer += 1;
    }
    return answer;
}
```
