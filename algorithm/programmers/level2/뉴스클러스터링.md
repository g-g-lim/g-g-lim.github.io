function solution(str1, str2) {
    let a = split(str1);
    let b = split(str2);
    const inter = intersection(a, b);
    const diff1 = diff(inter, a);
    const diff2 = diff(inter, b);
    return J(inter.length, inter.length + diff1.length + diff2.length);
}

function isAlphabet(char) {
    return /^[A-Za-z]$/.test(char);
  }

function split(str){
    const result = [];
    for (let i = 0; i < str.length - 1; i++) {
        const s = str.slice(i, i+2).split('')
        if (s.filter(isAlphabet).length === 2) {
            result.push(s.map(e => e.toUpperCase()).join(''))
        }
    }
    return result;
}


function intersection(str1, str2){
    let a, b;
    if (str1.length < str2.length) {
        a = [...str1];
        b = [...str2];
    } else {
        a = [...str2];
        b = [...str1];
    }

    const intersection = [];
    for (let i = 0; i < a.length; i++) {
        let e = a[i];
        for (let j = 0; j < b.length; j++) {
            if (e === b[j]) {
                intersection.push(e);
                b.splice(j, 1);
                break;
            }
        }
    }
    return intersection;
}

function diff(str1, str2) {
    let a, b;
    if (str1.length > str2.length) {
        a = [...str1];
        b = [...str2];
    } else {
        a = [...str2];
        b = [...str1];
    }

    const diff = [];
    for (let i = 0; i < a.length; i++) {
        let e = a[i];
        let found = false;
        for (let j = 0; j < b.length; j++) {
            if (e === b[j]) {
                found = true;
                b.splice(j, 1);
                break;
            }
        }
        if (!found) {
            diff.push(e);
        }
    }
    return diff;
}

function J(inter, union) {
    if (inter + union === 0) {
        return 65536;
    }
    return Math.floor((inter / union) * 65536)
}
