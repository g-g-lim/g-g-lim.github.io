# 과제진행하기

```python   
from datetime import datetime

def solution(plans):
    current_time = "00:00"

    completed = []
    plans = sorted(plans, key=lambda x: datetime.strptime(x[1], "%H:%M"))
    stops = []
    ing = None

    while current_time != "24:00":
        if ing is not None:
            ing[2] = int(ing[2]) - 1
            if ing[2] == 0:
                completed.append(ing[0])
                ing = None

        if len(plans) > 0:
            name, time, play = plans[0]
            if time == current_time:
                if ing is None:
                    ing = plans.pop(0)
                else:
                    stops.append(ing)
                    ing = plans.pop(0)
        if ing is None and len(stops) > 0:
            ing = stops.pop()

        if ing is None and len(stops) == 0 and len(plans) == 0:
            break

        current_time = add_time(current_time, 1)

    if ing is not None:
        completed.append(ing[0])

    if len(stops) > 0:
        while len(stops) > 0:
            stop = stops.pop()
            completed.append(stop[0])

    return completed

def add_time(time, minutes):
    hour, minute = map(int, time.split(":"))
    new_minute = minute + minutes
    if new_minute >= 60:
        quotient = new_minute // 60
        new_minute = new_minute % 60
        hour += quotient
    return f"{hour:02d}:{new_minute:02d}"
```
