# 무인도여행

```javascript
function solution(maps) {
    let answer = [];
    let visited = {};
    
    for (let i = 0; i < maps.length; i++) {
        for (let j = 0; j < maps[0].length; j++) {
            let sum = dfs([i,j], maps, visited);
            if (sum !== 0) {
                answer.push(sum);
            } else if (
                !visited[`${[i,j]}`] && 
                +maps[i][j] !== 0 &&
                 maps[i][j] !== 'X'
            ) {
                answer.push(+maps[i][j])
            }
        }
    }
    
    return answer.length === 0 ? [-1] : answer.sort((a, b) => a - b);
}

function dfs(start, maps, visited, cnt = 0) {
    if (maps[start[0]][start[1]] === 'X') {
        return 0;
    }

    for (const [i, j] of [[-1,0], [1,0], [0,1], [0,-1]]) {
        const next = [start[0] + i, start[1] + j];
        if (
            next[0] < 0 ||
            next[1] < 0 ||
            next[0] > maps.length - 1 ||
            next[1] > maps[0].length - 1
        ) {
            continue;
        }
        if (maps[next[0]][next[1]] === 'X') {
            continue;
        }
        if (visited[`${next}`]) {
            continue;
        }
        visited[`${next}`] = true;
        cnt += dfs(next, maps, visited, +maps[next[0]][next[1]])
    }

    return cnt;
}
```
