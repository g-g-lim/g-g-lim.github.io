# 숫자블록

```javascript
function solution(begin, end) {
    const answer = [];
    for (let i = begin; i <= end; i++) {
        if (i < 2) {
            answer.push(0);
            continue;
        }
        if (isPrime(i)) {
            answer.push(1);
            continue;
        }
        let n = 2;
        let resolved = false;
        let mod = undefined
        while (Math.sqrt(i) >= n) {
            if (i % n === 0) {
                mod = i / n;
                if (mod <= 10000000) {
                    answer.push(mod);
                    resolved = true;
                    break;
                } else {
                    mod = n;
                    n += 1;
                }
            } else {
                n += 1;
            }
        }
        if (!resolved) {
            answer.push(mod);
        }
    }
    return answer;
}

function isPrime(num) {
    if (num < 2) return false;
    if (num === 2) return true;
    if (num % 2 === 0) return false;
    const sqrt = Math.sqrt(num);
    for (let i = 3; i <= sqrt; i += 2) {
        if (num % i === 0) return false;
    }
    return true;
}
```
