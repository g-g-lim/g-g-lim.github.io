# 연속된부분수열합

<img src="/assets/images/programmers/연속된부분수열합.png" width="100%" alt="연속된부분수열합" />


```python   
def solution(sequence, k):
    slen = len(sequence)
    answer = [slen, slen]
    s, e = 0, 0
    m = slen
    ss = sequence[0]

    while s < slen and e < slen:
        if k == ss:
            if m > e - s + 1:
                answer = [s, e]
                m = e - s + 1
            elif m == e - s + 1:
                if answer[0] > s:
                    answer = [s, e]
            if s == e:
                if s == slen - 1 and e == slen - 1:
                    break
                e += 1
                ss += sequence[e]
            elif s < e:
                ss -= sequence[s]
                s += 1
        elif k > ss:
            if e < slen - 1:
                e += 1
                ss += sequence[e]
            else:
                ss -= sequence[s]
                s += 1
        else:  # k < ss
            if s == e:
                if s == slen - 1 and e == slen - 1:
                    break
                e += 1
                ss += sequence[e]
            elif s < e:
                ss -= sequence[s]
                s += 1
    return answer
```
