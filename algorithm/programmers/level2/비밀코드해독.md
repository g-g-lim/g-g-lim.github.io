# 비밀코드해독

<img src="/assets/images/비밀코드해독.png" width="100%" alt="비밀코드해독" />

```javascript   
function solution(n, q, ans) {
    const result = [];
    const combs = getCombinations(n);
    for (const comb of combs) {
        let can = true;
        for (let i = 0; i < q.length; i++) {
            if (ans[i] !== getMatchCount(comb, q[i])) {
                can = false;
                break;
            }
        }
        if (can) {
            result.push(comb);
        }
    }
    return result.length;
}

function getMatchCount(arr1, arr2) {
    let result = 0;
    for (const e1 of arr1) {
        for (const e2 of arr2) {
            if (e1 === e2) {
                result += 1;
                break;
            }
        }
    }
    return result;
}

function combinations(n) {
    const arr = Array.from({ length: n }, (e, idx) => idx + 1)
    const results = [];
    function combine(start, combo) {
        if (combo.length === 5) {
            results.push([...combo]);
            return;
        }
        for (let i = start; i < arr.length; i++) {
            combo.push(arr[i]);
            combine(i + 1, combo);
            combo.pop();
        }
    }
    combine(0, []);
    return results;
}
```
