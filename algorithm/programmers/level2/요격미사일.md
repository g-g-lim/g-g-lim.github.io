```python
def solution(targets):
    answer = 0
    targets = sorted(targets, key=lambda x: (x[0], x[1]))
    currIdx = 0
    remove = len(targets)
    while remove > 0:
        s1, e1 = targets[currIdx]
        remove -= 1
        for j in range(currIdx + 1, len(targets)):
            s2, e2 = targets[j]
            if (e1 <= s2 and e1 < e2) or (e2 <= s1 and s2 < s1):
                currIdx = j
                break
            s1, e1 = s2, min(e1, e2)
            remove -= 1
        answer += 1
    return answer
```
