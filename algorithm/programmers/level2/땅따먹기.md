# 땅따먹기

<img src="/assets/images/programmers/땅따먹기.png" width="100%" alt="땅따먹기" />

```javascript
function solution(land){
    if (land.length === 1) {
        return Math.max(...land[0]);
    }
    let dp = calculate(land[0], land[1]);
    for (let i = 2; i < land.length; i++) {
        dp = calculate(dp, land[i]);
    }
    return Math.max(...dp)
}

function calculate(arr1, arr2){
    const result = [0,0,0,0];

    for (let i = 0; i < arr1.length; i++) {
        for (const j of [0,1,2,3].filter((_, idx) => idx !== i)) {
            const sum = arr1[i] + arr2[j];
            if (result[j] < sum) {
                result[j] = sum;
            }
        }
    }
    return result;
}
```
