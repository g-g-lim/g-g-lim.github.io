# 리코쳇로봇

<img src="/assets/images/programmers/리코쳇로봇.png" width="100%" alt="리코챗로봇" />

```python
def solution(board):
    answer = 0
    start = None
    for i in range(len(board)):
        for j in range(len(board[i])):
            if board[i][j] == "R":
                start = (i, j)
                break

    queue = [(start, 0, set())]
    while queue:
        current, move, visited = queue.pop(0)
        i, j = current

        if board[i][j] == "G":
            return move
        for x, y in [[0, 1], [1, 0], [-1, 0], [0, -1]]:
            m, n = i, j
            while True:
                if (
                    m > len(board) - 1
                    or n > len(board[0]) - 1
                    or m < 0
                    or n < 0
                    or board[m][n] == "D"
                ):
                    m -= x
                    n -= y
                    break
                m += x
                n += y
            if m == i and n == j:
                continue
            if m > len(board) - 1 or n > len(board[0]) - 1 or m < 0 or n < 0:
                continue
            if (i, j, m, n) in visited or (m, n, i, j) in visited:
                continue
            visited.add((i, j, m, n))
            queue.append(((m, n), move + 1, visited))

    return answer or -1
```
