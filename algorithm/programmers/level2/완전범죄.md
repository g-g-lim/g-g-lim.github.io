# 완전범죄

<img src="/assets/images/완전범죄.png" width="100%" alt="완전범죄" />


```javascript   
function solution(info, n, m) {
    let bsum = info.map(e => e[1]).reduce((acc, a) => acc + a, 0);
    if (bsum < m) {
        return 0;
    }

    let answer = undefined;
    let visited = new Set();
    const dfs = (idx, asum, bsum) => {
        if (idx === info.length) {
            if (asum < n && bsum < m) {
                if (answer === undefined) {
                    answer = asum;
                } else {
                    answer = Math.min(answer, asum);
                }
            }
            return
        }
        if (asum >= n || bsum >= m) {
            return
        }
        if (answer !== undefined && asum > answer) {
            return
        }
        if (visited.has(idx + ',' + asum + ',' + bsum)) {
            return
        }
        visited.add(idx + ',' + asum + ',' + bsum);
        dfs(idx + 1, asum + info[idx][0], bsum);
        dfs(idx + 1, asum, bsum + info[idx][1]);
    }

    dfs(0, 0, 0);
    return answer ?? -1;
}
```
