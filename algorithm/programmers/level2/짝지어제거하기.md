# 짝지어제거하기

```javascript
function solution(s)
{
    const stack = [];
    for (let c of s) {
        if (stack.length === 0) {
            stack.push(c);
        } else if (stack[stack.length - 1] === c) {
            stack.pop();
        } else {
            stack.push(c);
        }
    }
    return stack.length === 0 ? 1 : 0
}
```
