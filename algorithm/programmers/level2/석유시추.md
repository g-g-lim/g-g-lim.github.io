# 석유시추

<img src="/assets/images/programmers/석유시추.png" width="100%" alt="석유시추" />


```python
def solution(land):
    answer = 0
    m = len(land)
    n = len(land[0])
    group = {"id": 1, "arr": [[0] * n for _ in range(m)]}

    for i in range(n):
        total = 0
        visited_groups = set()
        for j in range(m):
            if land[j][i] == 0:
                continue
            if group["arr"][j][i] in visited_groups:
                continue
            total += search((j, i), land, group)
            visited_groups.add(group["arr"][j][i])
        answer = max(answer, total)
    return answer

def search(start, land, group):
    if land[start[0]][start[1]] > 1:
        return land[start[0]][start[1]]
    total = 0
    queue = [start]
    visited = set()
    visited.add(start)
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    while queue:
        cur = queue.pop(0)
        if land[cur[0]][cur[1]] == 1:
            total += 1
            group["arr"][cur[0]][cur[1]] = group["id"]
        for d in directions:
            i = cur[0] + d[0]
            j = cur[1] + d[1]
            if i < 0 or j < 0 or i > len(land) - 1 or j > len(land[0]) - 1:
                continue
            if (i, j) in visited:
                continue
            if land[i][j]:
                visited.add((i, j))
                queue.append((i, j))
    if total > 1:
        for i, j in visited:
            land[i][j] = total
    group["id"] += 1
    return total
```
