# 호텔대실

```python
from datetime import datetime, timedelta
import heapq


def solution(book_time):
    parsed = []
    for e in book_time:
        start = datetime.strptime(e[0], "%H:%M")
        end = datetime.strptime(e[1], "%H:%M")
        if end.hour == 23 and end.minute >= 50:
            end = datetime.strptime(f"23:{end.minute + 59 - end.minute}", "%H:%M")
        else:
            end = datetime.strptime(e[1], "%H:%M")
            end = end + timedelta(minutes=10)
        parsed.append((end.time(), start.time()))

    book_time = sorted(parsed, key=lambda x: x[1])
    hq = [book_time[0]]
    answer = 1
    for i in range(1, len(book_time)):
        _, start = book_time[i]
        while len(hq) > 0:
            end, _ = hq[0]
            if start < end:
                heapq.heappush(hq, book_time[i])
                answer = max(answer, len(hq))
                break
            else:
                heapq.heappop(hq)
    return answer

```
