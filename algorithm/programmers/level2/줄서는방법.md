# 줄서는방법

<img src="/assets/images/줄서는방법.png" width="100%" alt="줄서는방법" />

```javascript
function solution(n, k) {
    function next(arr, kn) {
        const n = arr.length;
        const f = factorial(n);
        const mod = f / n;
    
        let rn = n - 1;
        let range = [f, f - mod];
        while (true) {
            if (range[0] >= kn && kn > range[1]) {
                break;
            }
            range = [range[1], range[1] - mod];
            rn -= 1;
        }
    
        return [arr[rn], range];
    }

    let arr = Array(n).fill(undefined).map((_, idx) => idx + 1);
    const result = [];

    while (result.length !== n){
        const re = next(arr, k);
        result.push(re[0]);
        arr = arr.filter(e => e !== re[0]);
        k -= re[1][1];
    }

    return result;
}

function factorial(n) {
    let result = 1;
    for (let i = 2; i <= n; i++) {
        result *= i;
    }
    return result;
}
```
