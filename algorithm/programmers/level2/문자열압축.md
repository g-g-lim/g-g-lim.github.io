# 문자열압축

<img src="/assets/images/문자열압축.png" width="100%" alt="문자열압축" />

```javascript
function solution(msg) {
    const answer = [];
    const dict = {};
    for (let i = 0; i < 26; i++) {
        dict[String.fromCharCode(65 + i)] = i + 1;
    }
    let lastValue = 27;

    for (let i = 0; i < msg.length; i++) {
        let from = i;
        let to = from + 1;
        let result = 0;
        while (to <= msg.length) {
            let s = msg.slice(from, to);
            if (dict[s] === undefined) {
                dict[s] = lastValue;
                lastValue += 1;
                break;
            } else {
                if (result !== 0) {
                    i += 1;
                }
                result = dict[s];
                to += 1;
            }
        }
        answer.push(result);
    }

    return answer;
}
```
