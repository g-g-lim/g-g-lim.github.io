# 광물캐기

```javascript
const solution = (picks, minerals) => {
    let min = Infinity;

    (function recursive(_picks, _minerals, fatigue) {
        if (_picks.filter(e => e === 0).length === 3 || _minerals.length === 0) {
            min = Math.min(min, fatigue);
            return;
        }


        for (let i = 0; i < _picks.length; i++) {
            const p = _picks[i];

            if (p === 0) {
                continue;
            }

            const [result, _remainMinerals] = mining(i === 0 ? 'diamond' : i === 1 ? 'iron' : 'stone', _minerals);

            recursive(_picks.map((e, idx) => idx === i ? e - 1 : e), _remainMinerals, fatigue + result);
        }

    })(picks, minerals, 0);

    return min;
}

const mining = (pickaxe, minerals) => {
    const fatigueTables = {
        diamond: {
            diamond: 1,
            iron: 5,
            stone: 25,
        },
        iron: {
            diamond: 1,
            iron: 1,
            stone: 5,
        },
        stone: {
            diamond: 1,
            iron: 1,
            stone: 1,
        },
    }

    let fatigue = 0;
    for (let i = 0; i < Math.min(5, minerals.length); i++) {
        fatigue += fatigueTables[minerals[i]][pickaxe];
    }

    return [fatigue, minerals.slice(5, minerals.length)]
}
```
