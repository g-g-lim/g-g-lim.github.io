# 퍼즐게임챌린지

<img src="/assets/images/programmers/퍼즐게임챌린지-1.png" width="100%" alt="퍼즐게임챌린지" />

```python
def brute_force(diffs, times, level):
    total = 0
    for i in range(len(diffs)):
        d = diffs[i]
        if d <= level:
            total += times[i]
        else:
            cur = (d - level + 1) * times[i]
            if i - 1 >= 0:
                cur += (d - level) * times[i - 1]
            total += cur
    return total


def solution(diffs, times, limit):
    left, right = 1, max(diffs)
    while left <= right:
        level = (left + right) // 2
        total = brute_force(diffs, times, level)
        if total <= limit:
            right = level - 1
            answer = level
        elif total > limit:
            left = level + 1
    return answer
```
